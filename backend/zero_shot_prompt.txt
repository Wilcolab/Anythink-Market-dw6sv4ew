import re

def to_snake_case(text):
    """
    Converts a given string to snake_case.

    Args:
        text (str): The input string.

    Returns:
        str: The converted snake_case string.
    """
    text = re.sub(r'(?<!^)(?=[A-Z])', '_', text).lower()
    text = re.sub(r'\s+', '_', text)
    return text
